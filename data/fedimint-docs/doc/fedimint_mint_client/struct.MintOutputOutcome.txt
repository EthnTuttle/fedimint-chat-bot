Converts an Option<String [../../std/string/struct.String.html]> into an Option<usize
[https://doc.rust-lang.org/1.67.0/std/primitive.usize.html]>, preserving the original. The map
[https://doc.rust-lang.org/1.67.0/core/option/enum.Option.html#method.map] method takes the self argument by value, consuming the

let text: Option<String> = Some("Hello, world!".to_string());
// First, cast `Option<String>` to `Option<&String>` with `as_ref`,
// then consume *that* with `map`, leaving `text` on the stack.
let text_length: Option<usize> = text.as_ref().map(|s| s.len());
println!("still can print text: {text:?}");
